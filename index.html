CREATE TABLE IF NOT EXISTS eventos ( id BIGSERIAL PRIMARY KEY, cpf TEXT NOT NULL, placa TEXT NOT NULL, associacao TEXT NOT NULL, categoria TEXT NOT NULL, tipo_evento TEXT NOT NULL, valor_fipe NUMERIC(12,2), cota NUMERIC(12,2), vidro BOOLEAN DEFAULT false, terceiros BOOLEAN DEFAULT false, reincidencia BOOLEAN DEFAULT false, observacoes TEXT, created_at TIMESTAMPTZ DEFAULT now() ); -- Índices para busca rápida CREATE INDEX idx_eventos_cpf ON eventos(cpf); CREATE INDEX idx_eventos_placa ON eventos(placa); CREATE INDEX idx_eventos_assoc ON eventos(associacao); CREATE INDEX idx_eventos_created ON eventos(created_at DESC); -- Row Level Security ALTER TABLE eventos ENABLE ROW LEVEL SECURITY; CREATE POLICY "Allow all" ON eventos FOR ALL USING (true);
